                    1

                    2
                                                                
                                          
                      Aggiungete il riso e fatelo tostare qualche minuto.
                      
                                                              
                                                
													
												
                                                
													
												
                                              
                                        
                                        
                    
                    
                    
                    
                
                                                
                  
                    3
                                                                
                                          
                      Sfumare con la birra.
                      
                                                              
                                                
													
												
                                              
                                        
                    
                    
                
                                                
                  
                    4

                    5

                    6
