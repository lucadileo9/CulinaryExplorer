media


                                                
                                                                      
                
                                          500 g fagioli in scatola
                                      
              
                                  
                
                                          150 g passata pomodoro
                                      
              
                                  
                
                                          3 salsicce
                                      
              
                                  
                
                                          1/2 bicchiere di vino
                                      
              
                                  
                
                                          3 spicchi d'aglio
                                      
              
                                  
                
                                          3 foglie di salvia
                                      
              
                                  
                
                                          olio extravergine di oliva
                                      
              
                                  
                
                                          sale
                                      
              
                                  
                
                                          peperoncino
                                      
              
                                                                                                Attrezzature
              
                
                                  Casseruola                                
              
                          

                    
                            Preparazione
                                
                  
                    1
                                                                
                                          
                      Mettete in una casseruola l'olio, il vino, l'aglio sbucciato, la salvia e le salsicce.
                      
                                                              
                                                
													
												
                                              
                                                            
                    
                    
                    
                
                                                
                  
                    2
                                                                
                                          
                      Fatele cuocere bene.
                      
                                                              
                                                
													
												
                                              
                                        
                                        
                    
                    
                    
                    
                
                                                
                  
                    3
                                                                
                                          
                      Foratele per far uscire il liquido in eccesso.

                      
                                          
                      Scolate i fagioli.
                      
                                                              
                                                
													
												
                                              
                                        
                    
                    
                
                                                
                  
                    4
                                                                
                                          
                      Aggiungeteli in casseruola.
                      
                                                              
                                                
													
												
                                              
                                        
                    
                    
                
                                                
                  
                    5
                                                                
                                          
                      Aggiungete il pomodoro, il sale e il peperoncino.

                      
                                          
                      ,
                      
                                                              
                                                
													
												
                                              
                                        
                    
                    
                
                                                
                  
                    6
                                                                
                                          
                      Cuocete per 25 minuti.

                      
                                          
                      :
                      
                                                              
                                                
													
												
                                              
                                        
                    
                    
                
                                                
                  
                    7
                                                                
                                          
                      Servite con pane casereccio.
